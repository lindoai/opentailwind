import { Client } from '@yourapp/sdk';

// Initialize the client
const client = new Client({
  apiKey: process.env.API_KEY,
  environment: 'production'
});

// Create a new resource
const result = await client.resources.create({
  name: 'My Resource',
  type: 'document',
  metadata: { priority: 1 }
});

console.log(result.id); // res_abc123

package main

import (
    "os"
    "github.com/yourapp/sdk-go"
)

func main() {
    // Initialize the client
    client := yourapp.NewClient(
        os.Getenv("API_KEY"),
        yourapp.Production,
    )

    // Create a new resource
    result, _ := client.Resources.Create(&yourapp.CreateParams{
        Name: "My Resource",
        Type: "document",
    })

    fmt.Println(result.ID) // res_abc123
}
